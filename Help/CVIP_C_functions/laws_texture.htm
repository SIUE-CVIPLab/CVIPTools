NAME

     laws_texture - calculates 15 laws texture features, these features are based on 
                    Level, Edges, Spots, Ripple and Waves. 
 
 

SYNOPSIS

      #include <math.h>
      #include "CVIPimage.h"
 
     double *laws_texture(Image* image, Image* labeled_image, int row, int col, int window_size, int vector1, int vector2)
 
      <image> - pointer to original image structure
      <label_image> - pointer to a labeled Image structure
      <row> - a row coordinate within the object of interest
      <column>  -  a  column  coordinate  within  the  object  of
       Interest
      <window_size> - a neighborhood size used in calculation 
       <vector1> - an int value used for selecting one of above feature
       <vector2> - an int value used for selecting one of above feature
 
 

PATH

     $CVIPtoolsHOME\CVIPC\FEATURE\laws_texture.c

DESCRIPTION

     This function calculates 15 laws texture features. These 15 features are combination of 
     Level, Edges, Spots, Ripple and Waves. Each feature calculates 5 of the histogram features
     Mean, Standard deviation, Skew, Energy and Entropy. For monochrome image each feature gives 
     5 histogram features. For color image each feature gives 5 histogram features for each 
     bands, totally gives 15 values for color image.
 
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

    An array of 15 feature values for color image for single feature.
    An array of 5 features vales for monochrome image for single feature.
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include "CVIPtoolkit.h"
      #include "CVIPobject.h"
      #include "CVIPconvert.h"
      #include <math.h>
      #include "ObjectContour.h"
 
      void main()
      {
         Image *inputImage, *labeledImage;
         IMAGE_FORMAT format;
         char *inputfile, *outputfile;
         int rows, cols,bands, r, c,i;
         double *ptr;
 
         setDisplay_Image("picture", "Default");
 
         print_CVIP("\nEnter the original image File Name:  ");
         inputfile = (char *) getString_CVIP();
         inputImage = read_Image(inputfile,1);
         view_Image(inputImage,inputfile);
         
         print_CVIP("\nEnter the masked image File Name:  ");
         inputfile = (char *) getString_CVIP();
         labeledImage = read_Image(inputfile,1);
         view_Image(labeledImage,inputfile);
 
 
         rows = getNoOfRows_Image(inputImage);
         cols = getNoOfCols_Image(inputImage);
         bands= getNoOfBands_Image(inputImage);
         print_CVIP("\nEnter the row coordinate of a point
         on the labled image: ");
         r = getInt_CVIP(10, 0, rows);
         print_CVIP("\nEnter the col coordinate of the point
         on the labled image: ");
         c = getInt_CVIP(10, 0, cols);
         /* the following call calculates the law texture
         features */
         ptr=laws_texture(inputImage,labeledImage,r,c,15,0,0);
         print_CVIP("The law texture features of the input
         image are:");
         /*code for retrieving feature values*/
         for(i=0;i<5;i++)//for each histogram feature(Mean,Standard deviation,
                         // Skew, Energy and Entropy)
          {
            for(k=0;k<bands;k++)//for each band
           {
             print_CVIP("law%d=%f\n",i+1,ptr[i+(k*5)]);
           }
         }  
          free(inputfile);
       }
